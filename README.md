<h3>Вызов плагина</h3>
```
$('.class').dragger();
```

<h3>Настройки</h3>
```
$('.dragger').dragger({
	starClass: 'dragger-start',
	endClass: 'dragger-end',
	dragClass: 'drag',
	drag: true
});
```

<h3>Html</h3>
```
<div class="dragger">
	<img src="dragger-start.png">
	<img src="dragger-end.png">
</div>
```
